# *********** Freigaben am Ende der Datei einfügen ***********
# sudo nano /etc/samba/smb.conf
# Die Namen Freigabe1, Freigabe2, Freigabe3 können durch eigene Namen ersetzt werden

[Freigabe1]
   comment = Raspberry Pi Freigabe
   path = /home/samba/shared1
   browseable = yes
   writeable = yes
   only guest = no
   create mask = 0775
   directory mask = 0775
   valid users = samba, [weitere User]
   force user = samba
   public = no
